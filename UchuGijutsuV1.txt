5. Memory Subsystem & Roofline Model

# ============================================================================= # Memory Subsystem & Roofline Model # NEW: Roofline-based bandwidth-limited throughput, SRAM tiling, # ECC bandwidth overhead, HBM multi-stack modelling. # ============================================================================= @dataclass class MemoryStats: hbm_bw_utilisation: float = 0.0 # 0..1 effective_tflops: float = 0.0 memory_bound: bool = True arithmetic_intensity: float = 0.0 # flops / byte (ridge point) cache_hit_rate: float = 0.0 class MemorySubsystem: """ Models HBM + on-chip SRAM with Roofline throughput limiting. Reference: Williams, Waterman, Patterson (2009) "Roofline: an insightful visual performance model for floating-point programs and multicore architectures." """ def __init__(self, cfg: MemoryConfig, compute_cfg: ComputeConfig): self.cfg = cfg self.compute_cfg = compute_cfg # Raw HBM bandwidth (bytes/s), reduced by ECC overhead ecc_overhead = (1 + cfg.ecc_overhead_pct / 100) if cfg.ecc_enabled else 1.0 self.hbm_bw = (cfg.hbm_bw_tb_s * 1e12) / ecc_overhead # On-chip SRAM size in bytes self.sram_size_bytes = cfg.onchip_sram_mb * 1024**2 # Effective bandwidth scales with number of HBM stacks self.effective_bw = self.hbm_bw * cfg.num_hbm_stacks logger.debug( f"[MemorySubsystem] HBM BW={format_si(self.effective_bw, 'B/s')} " f"SRAM={cfg.onchip_sram_mb} MB ECC={'on' if cfg.ecc_enabled else 'off'}" ) # ------------------------------------------------------------------ def _peak_compute_tflops(self, freq_ghz: float) -> float: """Peak array throughput in TFLOPS at given frequency.""" N = self.compute_cfg.array_size dtype = self.compute_cfg.dtype sparsity = self.compute_cfg.sparsity # ops/cycle = 2 * N^2 (each PE does 1 MAC = 2 ops) ops_per_cycle = 2 * N * N * dtype.relative_ops # Sparsity speedup (structured 2:4 gives ~2x at 0.5) sparsity_mult = 1.0 / (1.0 - min(sparsity, 0.9)) peak_ops_s = ops_per_cycle * freq_ghz * 1e9 * sparsity_mult return peak_ops_s / 1e12 def get_arithmetic_intensity(self, matrix_size: int) -> float: """ Arithmetic intensity (flops/byte) for a square GEMM of size M. Simplified: AI = M / (2 * dtype_bytes) assuming square matrices and ignoring output traffic. """ dtype_bytes = self.compute_cfg.dtype.bytes_per_elem return matrix_size / (2.0 * dtype_bytes) def fits_in_sram(self, matrix_size: int) -> bool: """Check whether the matrix pair fits in on-chip SRAM.""" dtype_bytes = self.compute_cfg.dtype.bytes_per_elem required = 2 * matrix_size * matrix_size * dtype_bytes return required <= self.sram_size_bytes def get_effective_throughput( self, matrix_size: int, freq_ghz: float = 1.5, ) -> MemoryStats: """ Apply Roofline model to determine actual achievable TFLOPS. Returns MemoryStats with roofline analysis. """ peak_compute = self._peak_compute_tflops(freq_ghz) ai = self.get_arithmetic_intensity(matrix_size) # Ridge point: AI where compute and memory bounds cross ridge_point = peak_compute * 1e12 / self.effective_bw # Roofline: min(memory-roof, compute-roof) memory_roof_tflops = (self.effective_bw * ai) / 1e12 effective_tflops = min(memory_roof_tflops, peak_compute) # Boost throughput if matrix fits in SRAM (cache-hit rate ~1) cache_hit = 1.0 if self.fits_in_sram(matrix_size) else 0.0 if cache_hit > 0: # On-chip bandwidth is ~10x HBM sram_bw = self.effective_bw * 10 memory_roof_sram = (sram_bw * ai) / 1e12 effective_tflops = min(memory_roof_sram, peak_compute) return MemoryStats( hbm_bw_utilisation=min(memory_roof_tflops / peak_compute, 1.0), effective_tflops=effective_tflops, memory_bound=(ai < ridge_point), arithmetic_intensity=ai, cache_hit_rate=cache_hit, ) def compute_duration_s( self, flops: float, matrix_size: int, freq_ghz: float = 1.5, ) -> float: """Return wall-clock seconds to complete `flops` operations.""" stats = self.get_effective_throughput(matrix_size, freq_ghz) tflops = max(stats.effective_tflops, 1e-9) return flops / (tflops * 1e12) 

6. Thermal & Power Model — DVFS v2

# ============================================================================= # Thermal & Power Model — DVFS v2 # NEW: # - Joint voltage-frequency scaling with process-node limits # - Clock distribution + interconnect power term # - Temperature + voltage dependent leakage # - Thermal cycling for eclipse/sunlight transitions # - scale_voltage_freq() method # ============================================================================= @dataclass class ThermalState: temperature_k: float = 300.0 power_w: float = 0.0 freq_ghz: float = 1.5 voltage_v: float = 0.9 throttled: bool = False class ThermalPowerModel: """ Realistic DVFS-aware power and thermal model for a space AI chip. Power breakdown: P_total = P_dynamic + P_leakage + P_clock_dist P_dynamic = base * dynamic_frac * (f/f_nom) * (V/V_nom)^2 P_leakage = base * static_frac * exp(k*(T-T_nom)/T_nom) * (V/V_nom)^3 P_clock = base * dynamic_frac * clock_dist_frac * (f/f_nom) """ def __init__(self, cfg: SimConfig): self.pw = cfg.power self.dvfs = cfg.dvfs self.node = cfg.compute.process_node self._state = ThermalState( temperature_k=cfg.orbital.T_nom_k if hasattr(cfg.orbital, "T_nom_k") else cfg.power.T_nom_k, freq_ghz=cfg.dvfs.nom_freq_ghz, voltage_v=cfg.dvfs.nom_voltage_v, ) # Thermal resistance of radiator (K/W); tuned to 400 mm² chip self.theta_jc = 0.8 # junction-to-case °C/W self.radiator_area_m2 = cfg.compute.area_mm2 * 1e-6 # same as die area (placeholder) # ------------------------------------------------------------------ # Core power calculation # ------------------------------------------------------------------ def power_at_state( self, temp_k: float, freq_ghz: float, voltage_v: float, utilisation: float = 1.0, ) -> float: """ Full power consumption in Watts. Args: temp_k: Junction temperature in Kelvin. freq_ghz: Operating frequency. voltage_v: Supply voltage. utilisation: 0..1 fraction of compute units active. """ pw = self.pw f_ratio = freq_ghz / self.dvfs.nom_freq_ghz v_ratio = voltage_v / self.dvfs.nom_voltage_v # Dynamic power (gates switching): P ∝ f * C * V² p_dynamic = (pw.base_power_w * pw.dynamic_frac * f_ratio * (v_ratio ** 2) * utilisation) # Leakage: temperature + voltage dependent delta_T = (temp_k - pw.T_nom_k) / pw.T_nom_k p_leakage = (pw.base_power_w * pw.static_frac * math.exp(pw.leakage_factor * delta_T) * (v_ratio ** 3)) # Clock distribution + interconnect (temperature-independent) p_clock = (pw.base_power_w * pw.dynamic_frac * pw.clock_dist_frac * f_ratio) return p_dynamic + p_leakage + p_clock # ------------------------------------------------------------------ # DVFS scaling with process-node limits # ------------------------------------------------------------------ def scale_voltage_freq( self, target_freq_ghz: float, utilisation: float = 1.0, ) -> ThermalState: """ Compute a new (V, f) operating point, respecting process node limits. Returns updated ThermalState. Voltage scaling: V = V_nom * sqrt(f / f_nom) (approximate DVFS law). """ f_max = self.node.max_freq_ghz() v_min = self.node.min_voltage() v_max = self.node.max_voltage() v_nom = self.dvfs.nom_voltage_v f_nom = self.dvfs.nom_freq_ghz # Clamp frequency to process limits f_actual = min(max(target_freq_ghz, 0.1), f_max) # Voltage follows frequency via approximate quadratic DVFS law if self.dvfs.voltage_scaling: v_actual = v_nom * math.sqrt(f_actual / f_nom) v_actual = min(max(v_actual, v_min), v_max) else: v_actual = v_nom # Re-solve temperature given new power power = self.power_at_state( self._state.temperature_k, f_actual, v_actual, utilisation ) temp_new = self._solve_temperature(power) throttled = temp_new >= self.pw.T_max_k if throttled: # Emergency throttle: halve frequency f_actual = f_actual / 2.0 v_actual = v_nom * math.sqrt(f_actual / f_nom) v_actual = min(max(v_actual, v_min), v_max) power = self.power_at_state(temp_new, f_actual, v_actual, utilisation) temp_new = self._solve_temperature(power) logger.warning( f"[DVFS] Thermal throttle! New f={f_actual:.2f} GHz " f"T={temp_new:.1f} K" ) self._state = ThermalState( temperature_k=temp_new, power_w=power, freq_ghz=f_actual, voltage_v=v_actual, throttled=throttled, ) return self._state # ------------------------------------------------------------------ # Thermal cycling (eclipse/sunlight for space) # ------------------------------------------------------------------ def apply_orbital_thermal_cycle( self, in_eclipse: bool, radiator_area_m2: float = 0.05, ) -> float: """ Compute steady-state temperature based on orbital phase. Space radiator power: P_rad = ε * σ * A * T^4 In eclipse: environment = 3 K (cosmic background) In sunlight: solar load ≈ 1361 W/m² * absorptance """ stefan_boltzmann = 5.67e-8 emissivity = 0.85 absorptance = 0.15 solar_flux = 1361.0 # W/m² at 1 AU P_chip = self.power_at_state( self._state.temperature_k, self._state.freq_ghz, self._state.voltage_v, ) if in_eclipse: T_env = 3.0 # deep space background P_solar = 0.0 else: T_env = 250.0 # sunlit face temperature estimate P_solar = absorptance * solar_flux * radiator_area_m2 # Energy balance: P_chip + P_solar = P_radiated + P_conducted # Simplified: T_chip^4 = (P_chip + P_solar) / (ε * σ * A) + T_env^4 P_total = P_chip + P_solar T_chip_k = (P_total / (emissivity * stefan_boltzmann * radiator_area_m2) + T_env ** 4) ** 0.25 self._state.temperature_k = T_chip_k return T_chip_k def _solve_temperature(self, power_w: float) -> float: """Simple resistive thermal model: T = T_ambient + R_th * P.""" T_ambient = 250.0 # K, nominal space ambient (cold plate) return T_ambient + self.theta_jc * power_w @property def state(self) -> ThermalState: return self._state 

7. Radiation Effects — RHBD v2

# ============================================================================= # Radiation Effects — RHBD v2 # NEW: # - Error injection during computation (not just at input) # - Multiple error types: SEU, SET (transient), SEL (latch-up risk) # - Different rates for logic vs memory regions # - Realistic ECC + scrubbing simulation # - reliability_score and SDC rate output # ============================================================================= from enum import auto class RadiationErrorType(Enum): SEU = auto() # Single Event Upset — bit-flip in memory/register SET = auto() # Single Event Transient — glitch in combinational logic SEL = auto() # Single Event Latch-up — potential destructive (mitigated by RHBD) @dataclass class RadiationStats: total_seu: int = 0 total_set: int = 0 total_sel: int = 0 corrected_by_ecc: int = 0 uncorrectable: int = 0 # multi-bit upsets sdc_rate: float = 0.0 # silent data corruption per hour reliability_score: float = 1.0 # 0..1 class RHBDv2: """ Radiation-Hardened-By-Design v2 error model. Models: - SEU in SRAM / register files / accumulator arrays - SET in MAC units and control logic - SEL risk with RHBD mitigation - ECC (SECDED) correction of single-bit upsets - Periodic SRAM scrubbing - TMR (optional) """ def __init__(self, cfg: SimConfig, orbital: "OrbitalEnvironment"): self.rad = cfg.radiation self.orbital = orbital self._stats = RadiationStats() self._last_scrub_time = 0.0 self._accumulated_time = 0.0 # RHBD attenuates base SEU rate self._rhbd_attenuation = cfg.radiation.rhbd_factor # ------------------------------------------------------------------ def _effective_seu_rate(self, num_bits: int) -> float: """Effective SEU events/s for a memory of `num_bits` bits.""" seu_per_bit_s = ( self.orbital.get_seu_rate() # environment scaling * self._rhbd_attenuation # RHBD mitigation * self.rad.seu_rate_base # process baseline ) return seu_per_bit_s * num_bits def inject_errors_at_input( self, tensor: torch.Tensor, duration_s: float, error_type: RadiationErrorType = RadiationErrorType.SEU, ) -> Tuple[torch.Tensor, int]: """ Inject radiation-induced bit-flip errors into a tensor. Returns (corrupted_tensor, num_errors_injected). """ flat = tensor.float().reshape(-1) num_bits = flat.numel() * 32 # assume FP32 repr for bit counting if error_type == RadiationErrorType.SEU: rate = self._effective_seu_rate(num_bits) elif error_type == RadiationErrorType.SET: rate = (self.rad.set_rate_base * self._rhbd_attenuation * self.orbital.get_seu_rate() * num_bits * 0.1) else: # SEL rate = (self.rad.sel_rate_base * self._rhbd_attenuation * self.orbital.get_seu_rate() * num_bits * 0.01) expected_errors = rate * duration_s # Vectorised Poisson sampling (torch) n_errors = int(torch.poisson(torch.tensor(max(expected_errors, 0.0))).item()) if n_errors == 0: return tensor, 0 # Inject errors into random positions indices = torch.randint(0, flat.numel(), (n_errors,)) # Bit-flip modelled as additive noise proportional to value magnitude noise = torch.zeros_like(flat) noise[indices] = flat[indices].abs().clamp(min=1e-6) * torch.sign( torch.randn(n_errors) ) flat_corrupted = flat + noise return flat_corrupted.reshape(tensor.shape).to(tensor.dtype), n_errors def inject_errors_during_compute( self, tensor: torch.Tensor, duration_s: float, error_type: str = "seu", ) -> Tuple[torch.Tensor, int]: """ Model error accumulation in MAC units and pipeline registers during a computation of `duration_s` seconds. """ etype = RadiationErrorType[error_type.upper()] result, n = self.inject_errors_at_input(tensor, duration_s, etype) if etype == RadiationErrorType.SEU: self._stats.total_seu += n elif etype == RadiationErrorType.SET: self._stats.total_set += n else: self._stats.total_sel += n # ECC correction if self.rad.ecc_correction and n > 0: # SECDED corrects all single-bit; leaves multi-bit multi_bit_prob = 0.02 # ~2% chance of MBU given an upset uncorrectable = int(n * multi_bit_prob) corrected = n - uncorrectable self._stats.corrected_by_ecc += corrected self._stats.uncorrectable += uncorrectable # TMR — triple modular redundancy if self.rad.tmr_enabled and n > 0: # TMR corrects all single errors; incurs 3x area/power self._stats.corrected_by_ecc += n - self._stats.uncorrectable self._stats.uncorrectable = 0 return result, n def scrub_sram(self, current_time_s: float, sram_bits: int) -> int: """ Periodic SRAM scrubbing. Returns number of corrected upsets. """ elapsed = current_time_s - self._last_scrub_time if elapsed < self.rad.scrub_interval_s: return 0 seu_rate = self._effective_seu_rate(sram_bits) upsets_since_last_scrub = int( torch.poisson(torch.tensor(seu_rate * elapsed)).item() ) corrected = upsets_since_last_scrub # scrubbing corrects all detected self._last_scrub_time = current_time_s self._stats.corrected_by_ecc += corrected logger.debug( f"[RHBD] Scrub at t={current_time_s:.1f}s: " f"corrected {corrected} upsets in {sram_bits/1e6:.1f} Mbits SRAM" ) return corrected def compute_reliability_score(self, elapsed_s: float) -> RadiationStats: """ Compute SDC rate and reliability score over `elapsed_s` seconds. """ if elapsed_s <= 0: return self._stats uncorr = self._stats.uncorrectable sdc_per_s = uncorr / elapsed_s self._stats.sdc_rate = sdc_per_s * 3600 # per hour # Reliability: fraction of compute that was error-free total_events = (self._stats.total_seu + self._stats.total_set + self._stats.total_sel) if total_events == 0: self._stats.reliability_score = 1.0 else: self._stats.reliability_score = max( 0.0, 1.0 - uncorr / max(total_events, 1) ) return self._stats @property def stats(self) -> RadiationStats: return self._stats 

8. Orbital Environment Simulation

# ============================================================================= # Orbital Environment Simulation # NEW: # - Variable SEU rate based on orbit altitude + type # - Solar flare events (Poisson burst model) # - Temperature cycling (eclipse vs sunlight) # - Vacuum / microgravity thermal-interface modelling # - CREME96 / SPENVIS-inspired SEU scaling # ============================================================================= @dataclass class OrbitalEvent: event_type: str # "solar_flare", "eclipse_entry", "eclipse_exit" time_s: float duration_s: float intensity: float = 1.0 class OrbitalEnvironment: """ Simulate the radiation and thermal environment for a given orbit. SEU scaling based on CREME96/SPENVIS empirical models: - LEO: trapped protons + galactic cosmic rays (GCR) - MEO: Van Allen belt dominates - GEO: GCR + solar energetic particles - Deep space: max GCR flux """ # Base SEU rate multipliers relative to LEO 400 km (normalised) _ORBIT_SEU_FACTORS: Dict[OrbitType, float] = { OrbitType.LEO: 1.0, OrbitType.MEO: 12.0, # ~Van Allen inner belt OrbitType.GEO: 4.5, OrbitType.DEEP: 8.0, } _SOLAR_SEU_FACTORS: Dict[SolarActivity, float] = { SolarActivity.LOW: 0.5, SolarActivity.MODERATE: 1.0, SolarActivity.HIGH: 2.5, SolarActivity.FLARE: 50.0, # large SPE event } def __init__(self, cfg: OrbitalConfig): self.cfg = cfg self._current_time_s = 0.0 self._in_eclipse = False self._flare_active = False self._flare_end_time = 0.0 self._event_log: List[OrbitalEvent] = [] # ------------------------------------------------------------------ def get_seu_rate(self) -> float: """ Return normalised SEU rate multiplier (relative to LEO baseline). Includes altitude scaling, solar activity, and flare bursts. """ orbit_factor = self._ORBIT_SEU_FACTORS.get(self.cfg.orbit_type, 1.0) # Altitude scaling within LEO: higher altitude ≈ more trapped protons if self.cfg.orbit_type == OrbitType.LEO: alt_factor = 1.0 + (self.cfg.altitude_km - 400) / 1600.0 orbit_factor *= max(0.3, alt_factor) solar_factor = self._SOLAR_SEU_FACTORS.get(self.cfg.solar_activity, 1.0) # Active solar flare burst if self._flare_active and self._current_time_s < self._flare_end_time: solar_factor *= self._SOLAR_SEU_FACTORS[SolarActivity.FLARE] return orbit_factor * solar_factor def advance_time(self, delta_s: float) -> List[OrbitalEvent]: """ Step the orbital simulation by `delta_s` seconds. Returns any new events that occurred in this interval. """ self._current_time_s += delta_s events: List[OrbitalEvent] = [] # Eclipse cycling period = self.cfg.orbit_period_s phase = (self._current_time_s % period) / period eclipse_threshold = self.cfg.eclipse_fraction was_eclipse = self._in_eclipse self._in_eclipse = phase < eclipse_threshold if not was_eclipse and self._in_eclipse: evt = OrbitalEvent("eclipse_entry", self._current_time_s, 0.0) events.append(evt) self._event_log.append(evt) logger.debug(f"[Orbital] Eclipse entry at t={self._current_time_s:.1f}s") elif was_eclipse and not self._in_eclipse: evt = OrbitalEvent("eclipse_exit", self._current_time_s, 0.0) events.append(evt) self._event_log.append(evt) logger.debug(f"[Orbital] Eclipse exit at t={self._current_time_s:.1f}s") # Solar flare Poisson arrival (mean 1 flare / 10 days for moderate activity) flare_rate = { SolarActivity.LOW: 1 / (30 * 86400), SolarActivity.MODERATE: 1 / (10 * 86400), SolarActivity.HIGH: 1 / (3 * 86400), SolarActivity.FLARE: 1 / (1 * 86400), }.get(self.cfg.solar_activity, 1e-6) if not self._flare_active: expected = flare_rate * delta_s if np.random.random() < expected: flare_dur = np.random.exponential(3600) # ~1 hr mean duration intensity = np.random.lognormal(mean=0, sigma=0.5) self._flare_active = True self._flare_end_time = self._current_time_s + flare_dur evt = OrbitalEvent("solar_flare", self._current_time_s, flare_dur, intensity) events.append(evt) self._event_log.append(evt) logger.warning( f"[Orbital] Solar flare! intensity={intensity:.2f} " f"duration={flare_dur/3600:.2f}h" ) else: if self._current_time_s >= self._flare_end_time: self._flare_active = False return events @property def in_eclipse(self) -> bool: return self._in_eclipse @property def current_time_s(self) -> float: return self._current_time_s def vacuum_thermal_correction(self, base_resistance_c_per_w: float) -> float: """ In vacuum, convective cooling is absent. Increase effective thermal resistance by ~20% to model vacuum + microgravity degradation of thermal interface material. """ return base_resistance_c_per_w * 1.20 

9. Compute Array — FP8/FP16/INT8 + Sparsity

# ============================================================================= # Compute Array — FP8 / FP16 / INT8 / INT4 + Structured Sparsity # NEW: # - Full dtype registry (FP8, FP16, BF16, INT8, INT4) # - Structured 2:4 sparsity with realistic speedup model # - Per-layer quantisation hooks # - TFLOPS measurement with bandwidth accounting # ============================================================================= @dataclass class WorkloadLayer: name: str layer_type: str # "linear", "conv2d", "attention", "embedding" M: int # batch * seq_len OR output_features K: int # input_features N: int # output_features (for conv: K_h*K_w*C_in) dtype: DType = DType.INT8 sparsity: float = 0.0 def flops(self) -> float: """FLOPs for one GEMM (2*M*K*N).""" sparsity_mult = 1.0 - self.sparsity return 2.0 * self.M * self.K * self.N * sparsity_mult def bytes_read(self) -> float: return (self.M * self.K + self.K * self.N) * self.dtype.bytes_per_elem def arithmetic_intensity(self) -> float: b = self.bytes_read() return self.flops() / b if b > 0 else 0.0 class ComputeArray: """ Systolic array PE mesh with dtype-aware throughput, structured sparsity, and roofline limiting. """ def __init__(self, cfg: SimConfig, memory: MemorySubsystem): self.cfg = cfg self.memory = memory self.compute = cfg.compute self._dvfs = ThermalPowerModel(cfg) self._thermal_state = self._dvfs.scale_voltage_freq( cfg.dvfs.nom_freq_ghz ) def _compute_tflops(self, dtype: DType, freq_ghz: float) -> float: N = self.compute.array_size ops_per_cycle = 2 * N * N * dtype.relative_ops sparsity_speedup = 1.0 / max(1.0 - self.compute.sparsity, 0.5) return ops_per_cycle * freq_ghz * 1e9 * sparsity_speedup / 1e12 def run_layer( self, layer: WorkloadLayer, current_time_s: float, in_eclipse: bool = False, radiation_model: Optional[RHBDv2] = None, ) -> Dict[str, Any]: """ Execute a single workload layer. Returns dict with: duration_s, tflops, power_w, temperature_k, errors_injected, memory_bound. """ freq = self._thermal_state.freq_ghz mem_stats = self.memory.get_effective_throughput(layer.M, freq) effective_tflops = mem_stats.effective_tflops flops = layer.flops() # Duration from roofline duration_s = flops / max(effective_tflops * 1e12, 1e6) # Update thermal model self._thermal_state = self._dvfs.scale_voltage_freq( freq, utilisation=0.9 ) # Apply orbital thermal cycling if in_eclipse: self._dvfs.apply_orbital_thermal_cycle( in_eclipse=True, radiator_area_m2=0.05, ) # Radiation error injection during computation errors_injected = 0 if radiation_model is not None: dummy = torch.zeros(1, layer.M) _, errors_injected = radiation_model.inject_errors_during_compute( dummy, duration_s, "seu" ) return { "layer": layer.name, "duration_s": duration_s, "tflops": effective_tflops, "peak_tflops": self._compute_tflops(layer.dtype, freq), "power_w": self._thermal_state.power_w, "temperature_k": self._thermal_state.temperature_k, "throttled": self._thermal_state.throttled, "errors_injected": errors_injected, "memory_bound": mem_stats.memory_bound, "arithmetic_intensity": mem_stats.arithmetic_intensity, "cache_hit_rate": mem_stats.cache_hit_rate, } 

10. Workload Runner

# ============================================================================= # Workload Runner # NEW: # - Real transformer / LLM layer simulation # - Bandwidth-limited duration in run_workload # - Radiation error injection throughout execution # - Orbital environment integration # - num_iterations auto-scales to a target runtime # ============================================================================= def build_transformer_workload( seq_len: int = 512, hidden_dim: int = 1024, num_heads: int = 16, num_layers: int = 12, ffn_mult: int = 4, batch_size: int = 1, dtype: DType = DType.INT8, sparsity: float = 0.0, ) -> List[WorkloadLayer]: """ Build a list of WorkloadLayer objects representing a transformer model. Covers: QKV projection, attention (SDPA), FFN, output projection. """ layers: List[WorkloadLayer] = [] d = hidden_dim ff = hidden_dim * ffn_mult B, S = batch_size, seq_len for i in range(num_layers): prefix = f"L{i:02d}" # QKV projection: [B*S, d] x [d, 3d] layers.append(WorkloadLayer(f"{prefix}/QKV", "linear", B*S, d, 3*d, dtype, sparsity)) # Attention scores: [B*H, S, d/H] x [B*H, d/H, S] head_dim = d // num_heads layers.append(WorkloadLayer(f"{prefix}/AttnScore", "attention", B*num_heads*S, head_dim, S, dtype, sparsity)) # Attention output: [B*H, S, S] x [B*H, S, head_dim] layers.append(WorkloadLayer(f"{prefix}/AttnOut", "attention", B*num_heads*S, S, head_dim, dtype, sparsity)) # Output projection: [B*S, d] x [d, d] layers.append(WorkloadLayer(f"{prefix}/OutProj", "linear", B*S, d, d, dtype, sparsity)) # FFN: two linear layers layers.append(WorkloadLayer(f"{prefix}/FFN1", "linear", B*S, d, ff, dtype, sparsity)) layers.append(WorkloadLayer(f"{prefix}/FFN2", "linear", B*S, ff, d, dtype, sparsity)) return layers @dataclass class WorkloadResult: total_duration_s: float total_flops: float total_energy_j: float avg_tflops: float avg_power_w: float peak_power_w: float avg_temperature_k: float peak_temperature_k: float total_errors_injected: int total_uncorrectable: int memory_bound_fraction: float reliability_score: float layer_results: List[Dict[str, Any]] sdc_rate_per_hr: float def run_workload( cfg: SimConfig, layers: List[WorkloadLayer], num_iterations: int = -1, target_runtime_s: float = 10.0, ) -> WorkloadResult: """ Execute a multi-layer workload on the simulated chip. Args: num_iterations: If -1, auto-scales to hit target_runtime_s. target_runtime_s: Desired wall-clock runtime for the sim (seconds). """ set_seed(cfg.random_seed) os.makedirs(cfg.output_dir, exist_ok=True) memory = MemorySubsystem(cfg.memory, cfg.compute) array = ComputeArray(cfg, memory) orbital = OrbitalEnvironment(cfg.orbital) rhbd = RHBDv2(cfg, orbital) # Auto-scale iterations if num_iterations <= 0: # Estimate single-pass duration dummy_duration = sum( memory.compute_duration_s(lay.flops(), lay.M, cfg.dvfs.nom_freq_ghz) for lay in layers ) num_iterations = max(1, int(target_runtime_s / max(dummy_duration, 1e-9))) logger.info( f"[run_workload] Auto-scaled iterations={num_iterations} " f"(single_pass={dummy_duration:.4f}s, target={target_runtime_s:.1f}s)" ) total_time = 0.0 total_flops = 0.0 total_energy = 0.0 peak_power = 0.0 peak_temp = 0.0 temps: List[float] = [] powers: List[float] = [] n_mem_bound = 0 layer_results_all: List[Dict] = [] sram_bits = int(cfg.memory.onchip_sram_mb * 1024**2 * 8) for it in range(num_iterations): for layer in layers: # Advance orbital clock events = orbital.advance_time(0.001) # Scrub SRAM periodically rhbd.scrub_sram(orbital.current_time_s, sram_bits) result = array.run_layer( layer, current_time_s=orbital.current_time_s, in_eclipse=orbital.in_eclipse, radiation_model=rhbd, ) dt = result["duration_s"] power = result["power_w"] energy = power * dt total_time += dt total_flops += layer.flops() total_energy += energy peak_power = max(peak_power, power) peak_temp = max(peak_temp, result["temperature_k"]) temps.append(result["temperature_k"]) powers.append(power) if result["memory_bound"]: n_mem_bound += 1 if it == 0: layer_results_all.append(result) if it % max(1, num_iterations // 10) == 0: logger.info( f"[run_workload] iter={it}/{num_iterations} " f"t={total_time:.3f}s energy={joules_to_kwh(total_energy)*1e6:.2f}µWh " f"T={np.mean(temps[-len(layers):]):.1f}K" ) rad_stats = rhbd.compute_reliability_score(total_time) avg_tflops = (total_flops / total_time / 1e12) if total_time > 0 else 0 return WorkloadResult( total_duration_s=total_time, total_flops=total_flops, total_energy_j=total_energy, avg_tflops=avg_tflops, avg_power_w=float(np.mean(powers)) if powers else 0.0, peak_power_w=peak_power, avg_temperature_k=float(np.mean(temps)) if temps else 0.0, peak_temperature_k=peak_temp, total_errors_injected=rad_stats.total_seu + rad_stats.total_set, total_uncorrectable=rad_stats.uncorrectable, memory_bound_fraction=n_mem_bound / max(len(layers) * num_iterations, 1), reliability_score=rad_stats.reliability_score, layer_results=layer_results_all, sdc_rate_per_hr=rad_stats.sdc_rate, ) 

11. Parameter Sweep & Optimization Engine

# ============================================================================= # Parameter Sweep & Optimization Engine # NEW: # - Automatic optimization via scipy.optimize.differential_evolution # - W&B Pareto sweeps # - Configurable objectives: tflops_per_watt, reliability, latency # ============================================================================= from scipy.optimize import differential_evolution, OptimizeResult from typing import Callable @dataclass class SweepResult: array_size: int area_mm2: float freq_ghz: float dtype: str sparsity: float orbit: str tflops: float power_w: float tflops_per_watt: float reliability: float sdc_rate_per_hr: float temperature_k: float memory_bound_frac: float energy_j: float def _build_cfg_from_params( array_size: int, area_mm2: float, freq_ghz: float, dtype: DType, sparsity: float, orbit: OrbitType = OrbitType.LEO, base_cfg: Optional[SimConfig] = None, ) -> SimConfig: cfg = base_cfg or SimConfig() cfg.compute.array_size = array_size cfg.compute.area_mm2 = area_mm2 cfg.compute.dtype = dtype cfg.compute.sparsity = sparsity cfg.dvfs.nom_freq_ghz = freq_ghz cfg.orbital.orbit_type = orbit return cfg def run_parameter_sweep( array_sizes: List[int] = (128, 256, 512), area_mm2s: List[float] = (200, 400, 800), freq_ghzs: List[float] = (1.0, 1.5, 2.0), dtypes: List[DType] = (DType.INT8, DType.FP16), sparsities: List[float] = (0.0, 0.5), orbits: List[OrbitType] = (OrbitType.LEO, OrbitType.GEO), num_layers: int = 4, num_iterations: int = 1, base_cfg: Optional[SimConfig] = None, use_wandb: bool = False, ) -> List[SweepResult]: """ Grid sweep over design parameters. Returns a list of SweepResult objects — use plot_design_space() to visualise. """ results: List[SweepResult] = [] if use_wandb: try: import wandb wandb.init(project="uchugijutsu_sweep", name="parameter_sweep") except ImportError: use_wandb = False logger.warning("[Sweep] wandb not installed; disabling.") total = (len(array_sizes) * len(area_mm2s) * len(freq_ghzs) * len(dtypes) * len(sparsities) * len(orbits)) logger.info(f"[Sweep] Starting {total} configurations …") count = 0 for arr in array_sizes: for area in area_mm2s: for freq in freq_ghzs: for dtype in dtypes: for spar in sparsities: for orbit in orbits: cfg = _build_cfg_from_params( arr, area, freq, dtype, spar, orbit, base_cfg ) layers = build_transformer_workload( num_layers=num_layers, dtype=dtype, sparsity=spar, ) res = run_workload(cfg, layers, num_iterations=num_iterations) sr = SweepResult( array_size=arr, area_mm2=area, freq_ghz=freq, dtype=dtype.label, sparsity=spar, orbit=orbit.name, tflops=res.avg_tflops, power_w=res.avg_power_w, tflops_per_watt=res.avg_tflops / max(res.avg_power_w, 1e-6), reliability=res.reliability_score, sdc_rate_per_hr=res.sdc_rate_per_hr, temperature_k=res.avg_temperature_k, memory_bound_frac=res.memory_bound_fraction, energy_j=res.total_energy_j, ) results.append(sr) count += 1 if use_wandb: import wandb wandb.log(asdict(sr)) if count % 20 == 0: logger.info( f"[Sweep] {count}/{total} " f"arr={arr} f={freq:.1f}GHz dtype={dtype.label} " f"TFLOPS/W={sr.tflops_per_watt:.3f} " f"rel={sr.reliability:.3f}" ) if use_wandb: import wandb wandb.finish() return results def optimize_design( objective: str = "tflops_per_watt", constraints: Optional[Dict[str, float]] = None, base_cfg: Optional[SimConfig] = None, num_layers: int = 4, maxiter: int = 50, popsize: int = 8, seed: int = 42, ) -> Tuple[SweepResult, OptimizeResult]: """ Automatic design-space optimization using differential evolution. dtype index (0.0, 0.75), # sparsity ] def _objective(x: np.ndarray) -> float: arr_size = int(round(x[0] / 64) * 64) area = float(x[1]) freq = float(x[2]) dtype = dtype_list[int(round(x[3]))] sparsity = float(np.clip(x[4], 0, 0.75)) cfg = _build_cfg_from_params(arr_size, area, freq, dtype, sparsity, base_cfg=base_cfg) layers = build_transformer_workload(num_layers=num_layers, dtype=dtype, sparsity=sparsity) with warnings.catch_warnings(): warnings.simplefilter("ignore") res = run_workload(cfg, layers, num_iterations=1) # Constraint penalties penalty = 0.0 if "min_tflops" in constraints: deficit = constraints["min_tflops"] - res.avg_tflops if deficit > 0: penalty += deficit * 1000 if "max_power_w" in constraints: excess = res.avg_power_w - constraints["max_power_w"] if excess > 0: penalty += excess * 100 if "max_temp_k" in constraints: excess = res.avg_temperature_k - constraints["max_temp_k"] if excess > 0: penalty += excess * 50 if "min_reliability" in constraints: deficit = constraints["min_reliability"] - res.reliability_score if deficit > 0: penalty += deficit * 1000 if objective == "tflops_per_watt": score = -res.avg_tflops / max(res.avg_power_w, 1e-6) elif objective == "tflops": score = -res.avg_tflops elif objective == "reliability": score = -res.reliability_score elif objective == "latency": score = res.total_duration_s else: raise ValueError(f"Unknown objective: {objective}") return score + penalty logger.info( f"[Optimize] Starting differential_evolution for '{objective}' " f"(maxiter={maxiter}, popsize={popsize}) …" ) opt_result = differential_evolution( _objective, bounds=bounds, maxiter=maxiter, popsize=popsize, seed=seed, tol=1e-4, callback=lambda xk, convergence: logger.debug(f" [DE] convergence={convergence:.4f}"), ) # 13. Monte-Carlo Reliability Simulation

# ============================================================================= # Monte-Carlo Reliability Simulation # NEW: # - Full mission-duration MC simulation # - SDC accumulation model # - Reliability score distribution # - Expected mission lifetime under radiation budget # ============================================================================= @dataclass class MCReliabilityResult: mean_reliability: float std_reliability: float ci_low: float ci_high: float mean_sdc_per_hr: float p99_sdc_per_hr: float expected_mission_lifetime_h: float num_samples: int def monte_carlo_reliability( cfg: SimConfig, num_samples: int = 200, mission_duration_h: float = 8760.0, # 1 year sdc_budget_per_hr: float = 1e-4, layers: Optional[List[WorkloadLayer]] = None, ) -> MCReliabilityResult: """ Monte-Carlo simulation of chip reliability over a mission. Varies SEU rate, solar activity, and temperature stochastically across `num_samples` independent runs. Returns distribution of reliability scores and SDC rates. """ set_seed(cfg.random_seed) if layers is None: layers = build_transformer_workload(num_layers=4, dtype=cfg.compute.dtype) reliability_scores: List[float] = [] sdc_rates: List[float] = [] solar_activities = list(SolarActivity) solar_weights = [0.25, 0.40, 0.25, 0.10] # LOW, MOD, HIGH, FLARE logger.info( f"[MC] Starting {num_samples} reliability samples " f"(mission={mission_duration_h:.0f}h) …" ) for s in range(num_samples): # Sample random solar activity activity = np.random.choice(solar_activities, p=solar_weights) cfg_sample = SimConfig( memory=cfg.memory, power=cfg.power, dvfs=cfg.dvfs, radiation=cfg.radiation, compute=cfg.compute, orbital=OrbitalConfig( orbit_type=cfg.orbital.orbit_type, altitude_km=cfg.orbital.altitude_km + np.random.uniform(-50, 50), solar_activity=activity, ), sim_duration_s=min(mission_duration_h * 3600, 3600), # 1-hr slice random_seed=cfg.random_seed + s, ) res = run_workload(cfg_sample, layers, num_iterations=1) reliability_scores.append(res.reliability_score) sdc_rates.append(res.sdc_rate_per_hr) if (s + 1) % (num_samples // 5) == 0: logger.info( f"[MC] {s+1}/{num_samples} " f"mean_rel={np.mean(reliability_scores):.4f} " f"mean_sdc={np.mean(sdc_rates):.3e}/hr" ) rel_arr = np.array(reliability_scores) sdc_arr = np.array(sdc_rates) ci_lo, ci_hi = bootstrap_ci(rel_arr) # Estimate lifetime: time until cumulative SDC exceeds budget # P(failure at hour h) = 1 - (1 - mean_sdc)^h (geometric approx) mean_sdc = float(np.mean(sdc_arr)) if mean_sdc > 0: lifetime_h = math.log(0.5) / math.log(1 - min(mean_sdc, 1 - 1e-9)) else: lifetime_h = float("inf") return MCReliabilityResult( mean_reliability=float(np.mean(rel_arr)), std_reliability=float(np.std(rel_arr)), ci_low=ci_lo, ci_high=ci_hi, mean_sdc_per_hr=mean_sdc, p99_sdc_per_hr=float(np.percentile(sdc_arr, 99)), expected_mission_lifetime_h=lifetime_h, num_samples=num_samples, )
