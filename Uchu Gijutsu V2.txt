5. Memory Subsystem & Roofline Model

# ============================================================================= # Memory Subsystem & Roofline Model # (Unchanged from v1 — no improvements required here) # ============================================================================= @dataclass class MemoryStats: hbm_bw_utilisation: float = 0.0 effective_tflops: float = 0.0 memory_bound: bool = True arithmetic_intensity: float = 0.0 cache_hit_rate: float = 0.0 class MemorySubsystem: """ Models HBM + on-chip SRAM with Roofline throughput limiting. Reference: Williams, Waterman, Patterson (2009) "Roofline: an insightful visual performance model for floating-point programs and multicore architectures." """ def __init__(self, cfg: MemoryConfig, compute_cfg: ComputeConfig): self.cfg = cfg self.compute_cfg = compute_cfg ecc_overhead = (1 + cfg.ecc_overhead_pct / 100) if cfg.ecc_enabled else 1.0 self.hbm_bw = (cfg.hbm_bw_tb_s * 1e12) / ecc_overhead self.sram_size_bytes = cfg.onchip_sram_mb * 1024 ** 2 self.effective_bw = self.hbm_bw * cfg.num_hbm_stacks logger.debug( f"[MemorySubsystem] HBM BW={format_si(self.effective_bw, 'B/s')} " f"SRAM={cfg.onchip_sram_mb} MB ECC={'on' if cfg.ecc_enabled else 'off'}" ) def _peak_compute_tflops(self, freq_ghz: float) -> float: N = self.compute_cfg.array_size dtype = self.compute_cfg.dtype sparsity = self.compute_cfg.sparsity ops_per_cycle = 2 * N * N * dtype.relative_ops # v2: realistic 2:4 structured sparsity speedup (not just 1/(1-s)) sparsity_mult = _sparsity_speedup(sparsity) return ops_per_cycle * freq_ghz * 1e9 * sparsity_mult / 1e12 def get_arithmetic_intensity(self, matrix_size: int) -> float: return matrix_size / (2.0 * self.compute_cfg.dtype.bytes_per_elem) def fits_in_sram(self, matrix_size: int) -> bool: dtype_bytes = self.compute_cfg.dtype.bytes_per_elem return 2 * matrix_size * matrix_size * dtype_bytes <= self.sram_size_bytes def get_effective_throughput( self, matrix_size: int, freq_ghz: float = 1.5 ) -> MemoryStats: peak_compute = self._peak_compute_tflops(freq_ghz) ai = self.get_arithmetic_intensity(matrix_size) ridge_point = peak_compute * 1e12 / self.effective_bw memory_roof = (self.effective_bw * ai) / 1e12 eff_tflops = min(memory_roof, peak_compute) cache_hit = 1.0 if self.fits_in_sram(matrix_size) else 0.0 if cache_hit > 0: sram_bw = self.effective_bw * 10 eff_tflops = min((sram_bw * ai) / 1e12, peak_compute) return MemoryStats( hbm_bw_utilisation = min(memory_roof / peak_compute, 1.0), effective_tflops = eff_tflops, memory_bound = (ai < ridge_point), arithmetic_intensity = ai, cache_hit_rate = cache_hit, ) def compute_duration_s( self, flops: float, matrix_size: int, freq_ghz: float = 1.5 ) -> float: stats = self.get_effective_throughput(matrix_size, freq_ghz) return flops / max(stats.effective_tflops * 1e12, 1e6) def _sparsity_speedup(sparsity: float) -> float: """ v2: Realistic structured 2:4 sparsity speedup model. At s=0.50: ~1.85× (not the naive 2.0× — accounts for overhead). At s=0.75: ~2.8× (close to theoretical 4×, with overhead). Below 0.50: linear interpolation (partial benefit). Above 0.75: capped (hardware limitation). """ s = float(np.clip(sparsity, 0.0, 0.75)) if s <= 0.0: return 1.0 # Reference points: (0.5, 1.85), (0.75, 2.80) with overhead penalty if s <= 0.5: return 1.0 + (s / 0.5) * 0.85 else: return 1.85 + ((s - 0.5) / 0.25) * 0.95 

6. Thermal & Power Model — DVFS v2

# ============================================================================= # Thermal & Power Model — DVFS v2 # v2 Improvements: # - Thermal capacitance: dT/dt = (P - P_radiated) / C # - Radiator view factor + mission-lifetime degradation # - Temperature-dependent frequency throttle *curve* (not hard cut) # - Interconnect power scales with array_size # - Base power scales with area and process node (via SimConfig.scaled_base_power_w) # ============================================================================= @dataclass class ThermalState: temperature_k: float = 300.0 power_w: float = 0.0 freq_ghz: float = 1.5 voltage_v: float = 0.9 throttled: bool = False class ThermalPowerModel: """ DVFS-aware power and thermal model for a space AI chip with thermal inertia. Power breakdown: P_dynamic = P_base * dyn_frac * (f/f_nom) * (V/V_nom)² P_leakage = P_base * static_frac * exp(k*(T-T_nom)/T_nom) * (V/V_nom)³ P_clock = P_base * dyn_frac * clock_dist_frac * (f/f_nom) P_interconnect= P_base * dyn_frac * interconnect_frac * (f/f_nom) * array_ratio Thermal (with inertia): dT/dt = (P_chip + P_solar - P_radiated) / C_thermal """ def __init__(self, cfg: SimConfig): self.pw = cfg.power self.dvfs = cfg.dvfs self.node = cfg.compute.process_node self.tc = cfg.thermal_cap self._base_pw = cfg.scaled_base_power_w() # v2: area + node scaled self._array_ratio = (cfg.compute.array_size / cfg.power.ref_array_size) ** 2 self._state = ThermalState( temperature_k = cfg.power.T_nom_k, freq_ghz = cfg.dvfs.nom_freq_ghz, voltage_v = cfg.dvfs.nom_voltage_v, ) self._current_temp_k = cfg.power.T_nom_k # continuous thermal state # ------------------------------------------------------------------ def power_at_state( self, temp_k: float, freq_ghz: float, voltage_v: float, utilisation: float = 1.0, ) -> float: pw = self.pw f_ratio = freq_ghz / self.dvfs.nom_freq_ghz v_ratio = voltage_v / self.dvfs.nom_voltage_v p_dyn = self._base_pw * pw.dynamic_frac * f_ratio * (v_ratio ** 2) * utilisation delta_T = (temp_k - pw.T_nom_k) / pw.T_nom_k p_leak = (self._base_pw * pw.static_frac * math.exp(pw.leakage_factor * delta_T) * (v_ratio ** 3)) p_clock = self._base_pw * pw.dynamic_frac * pw.clock_dist_frac * f_ratio # v2: interconnect power scales with array² (data movement) p_ic = (self._base_pw * pw.dynamic_frac * pw.interconnect_frac * f_ratio * self._array_ratio) return p_dyn + p_leak + p_clock + p_ic # ------------------------------------------------------------------ def _throttled_freq(self, temp_k: float, target_freq_ghz: float) -> float: """ v2: Smooth throttle curve instead of hard cutoff. Above throttle_onset_k, frequency reduces linearly with temperature. """ onset = self.dvfs.throttle_onset_k if temp_k <= onset: return target_freq_ghz excess = temp_k - onset reduction = self.dvfs.throttle_slope * excess f_throttled = target_freq_ghz * max(0.1, 1.0 - reduction) logger.debug(f"[DVFS] Throttle curve: T={temp_k:.1f}K → f={f_throttled:.2f}GHz") return f_throttled def scale_voltage_freq( self, target_freq_ghz: float, utilisation: float = 1.0, ) -> ThermalState: f_max = self.node.max_freq_ghz() v_nom = self.dvfs.nom_voltage_v f_nom = self.dvfs.nom_freq_ghz # Apply smooth thermal throttle f_desired = self._throttled_freq(self._current_temp_k, target_freq_ghz) f_actual = min(max(f_desired, 0.1), f_max) throttled = f_actual < target_freq_ghz * 0.99 if self.dvfs.voltage_scaling: v_actual = v_nom * math.sqrt(f_actual / f_nom) v_actual = min(max(v_actual, self.node.min_voltage()), self.node.max_voltage()) else: v_actual = v_nom power = self.power_at_state(self._current_temp_k, f_actual, v_actual, utilisation) temp_new = self._solve_temperature_static(power) self._state = ThermalState( temperature_k = temp_new, power_w = power, freq_ghz = f_actual, voltage_v = v_actual, throttled = throttled, ) self._current_temp_k = temp_new return self._state # ------------------------------------------------------------------ def step_thermal( self, dt_s: float, in_eclipse: bool, radiator_area_m2: float = 0.05, ) -> float: """ v2: Integrate dT/dt = (P_chip + P_solar - P_radiated) / C_thermal over a timestep of dt_s seconds. Updates self._current_temp_k. Returns new temperature in K. """ tc = self.tc P_chip = self._state.power_w # Solar load if in_eclipse: P_solar = 0.0 else: solar_flux = 1361.0 # W/m² at 1 AU P_solar = tc.absorptance * solar_flux * radiator_area_m2 # Radiator emission: P_rad = ε * σ * view_factor_effective * A * T^4 view_factor = tc.radiator_view_factor * (1.0 - tc.radiator_degradation * tc.mission_year) view_factor = max(view_factor, 0.1) T = self._current_temp_k P_rad = tc.emissivity * 5.67e-8 * view_factor * radiator_area_m2 * (T ** 4) # Thermal inertia ODE (Euler step) dT_dt = (P_chip + P_solar - P_rad) / max(tc.capacitance_j_per_k, 1e-9) T_new = T + dT_dt * dt_s T_new = max(T_new, 3.0) # can't go below deep space background self._current_temp_k = T_new return T_new # ------------------------------------------------------------------ def apply_orbital_thermal_cycle( self, in_eclipse: bool, radiator_area_m2: float = 0.05 ) -> float: """ Convenience wrapper — step thermal model by 1 s. For backwards compatibility. """ return self.step_thermal(1.0, in_eclipse, radiator_area_m2) def _solve_temperature_static(self, power_w: float) -> float: """Fallback: simple resistive thermal model for startup.""" theta_jc = 0.8 # K/W T_ambient = 250.0 return T_ambient + theta_jc * power_w @property def state(self) -> ThermalState: return self._state

Does not mutate internal state. """ if elapsed_s <= 0: return RadiationStats() sdc_per_s = self._uncorr / elapsed_s sdc_rate = sdc_per_s * 3600 total_events = self._total_seu + self._total_set + self._total_sel reliability = (max(0.0, 1.0 - self._uncorr / max(total_events, 1)) if total_events > 0 else 1.0) return RadiationStats( total_seu = self._total_seu, total_set = self._total_set, total_sel = self._total_sel, corrected_by_ecc = self._ecc_corr, uncorrectable = self._uncorr, sdc_rate = sdc_rate, reliability_score = reliability, ) @property def stats(self) -> RadiationStats: """Current (mutable) stats snapshot — same shape as compute_reliability_score.""" return RadiationStats( total_seu = self._total_seu, total_set = self._total_set, total_sel = self._total_sel, corrected_by_ecc = self._ecc_corr, uncorrectable = self._uncorr, ) # Backwards-compatible alias RHBDv2 = RHBDv3 

8. Orbital Environment Simulation

# ============================================================================= # Orbital Environment Simulation # v2 Improvements: # - advance_time() is now called once per *iteration* (not per layer) # - orbital_advance_per_iter computed from expected duration in run_workload # - Flare and eclipse state unchanged (correct) # ============================================================================= @dataclass class OrbitalEvent: event_type: str # "solar_flare", "eclipse_entry", "eclipse_exit" time_s: float duration_s: float intensity: float = 1.0 class OrbitalEnvironment: """ Simulate radiation and thermal environment for a given orbit. """ _ORBIT_SEU_FACTORS: Dict[OrbitType, float] = { OrbitType.LEO: 1.0, OrbitType.MEO: 12.0, OrbitType.GEO: 4.5, OrbitType.DEEP: 8.0, } _SOLAR_SEU_FACTORS: Dict[SolarActivity, float] = { SolarActivity.LOW: 0.5, SolarActivity.MODERATE: 1.0, SolarActivity.HIGH: 2.5, SolarActivity.FLARE: 50.0, } def __init__(self, cfg: OrbitalConfig, rng: Optional[np.random.Generator] = None): self.cfg = cfg self._rng = rng or np.random.default_rng() self._current_time_s = 0.0 self._in_eclipse = False self._flare_active = False self._flare_end_time = 0.0 self._event_log: List[OrbitalEvent] = [] def get_seu_rate(self) -> float: orbit_factor = self._ORBIT_SEU_FACTORS.get(self.cfg.orbit_type, 1.0) if self.cfg.orbit_type == OrbitType.LEO: alt_factor = 1.0 + (self.cfg.altitude_km - 400) / 1600.0 orbit_factor *= max(0.3, alt_factor) solar_factor = self._SOLAR_SEU_FACTORS.get(self.cfg.solar_activity, 1.0) if self._flare_active and self._current_time_s < self._flare_end_time: solar_factor *= self._SOLAR_SEU_FACTORS[SolarActivity.FLARE] return orbit_factor * solar_factor def advance_time(self, delta_s: float) -> List[OrbitalEvent]: """ Step simulation by delta_s seconds. v2: Called once per *iteration* (not per layer) to avoid drift. # ============================================================================= # Monte-Carlo Reliability Simulation # v2 Improvements: # - Parallel execution via joblib # - Isolated RNG per sample (no shared state between workers) # - Weibull failure model for mission lifetime (replaces geometric approx) # - seed management: sample i gets seed = base + i # ============================================================================= try: import joblib as _joblib _MC_PARALLEL = True except ImportError: _MC_PARALLEL = False @dataclass class MCReliabilityResult: mean_reliability: float std_reliability: float ci_low: float ci_high: float mean_sdc_per_hr: float p99_sdc_per_hr: float expected_mission_lifetime_h: float num_samples: int lifetime_model: str = "weibull" def _mc_single_sample( s: int, cfg: SimConfig, layers: List[WorkloadLayer], solar_activities: List[SolarActivity], solar_weights: List[float], base_seed: int, ) -> Tuple[float, float]: """Single Monte-Carlo sample — executed in parallel worker.""" rng = np.random.default_rng(base_seed + s) activity = rng.choice(solar_activities, p=solar_weights) # type: ignore import copy cfg_s = copy.deepcopy(cfg) cfg_s.orbital.solar_activity = activity cfg_s.orbital.altitude_km += float(rng.uniform(-50, 50)) cfg_s.random_seed = base_seed + s res = run_workload(cfg_s, layers, num_iterations=1, rng=rng) return res.reliability_score, res.sdc_rate_per_hr def monte_carlo_reliability( cfg: SimConfig, num_samples: int = 200, mission_duration_h: float = 8760.0, sdc_budget_per_hr: float = 1e-4, layers: Optional[List[WorkloadLayer]] = None, n_jobs: int = -1, ) -> MCReliabilityResult: """ v2: Monte-Carlo simulation with joblib parallelism and Weibull lifetime model. Each sample gets an isolated RNG (base_seed + sample_index). No global state mutation between samples. """ if layers is None: layers = build_transformer_workload(num_layers=4, dtype=cfg.compute.dtype) solar_activities = list(SolarActivity) solar_weights = [0.25, 0.40, 0.25, 0.10] # LOW, MOD, HIGH, FLARE base_seed = cfg.random_seed logger.info(f"[MC] Starting {num_samples} samples (n_jobs={n_jobs}) …") if _MC_PARALLEL and n_jobs != 1: pairs = _joblib.Parallel(n_jobs=n_jobs, prefer="threads")( _joblib.delayed(_mc_single_sample)( s, cfg, layers, solar_activities, solar_weights, base_seed ) for s in range(num_samples) ) else: pairs = [ _mc_single_sample(s, cfg, layers, solar_activities, solar_weights, base_seed) for s in range(num_samples) ] rel_arr = np.array([p[0] for p in pairs]) sdc_arr = np.array([p[1] for p in pairs]) rng_ci = np.random.default_rng(base_seed + 9999) ci_lo, ci_hi = bootstrap_ci(rel_arr, rng=rng_ci) # v2: Weibull lifetime model # Assume failures follow Weibull(shape=1.5, scale=η) where η is chosen # so that P(T > t_50) = 0.5 under mean SDC rate. mean_sdc = float(np.mean(sdc_arr)) if mean_sdc > 0: # Simple exponential approximation (Weibull with shape=1 collapses to this) # For shape > 1 (wear-out), lifetime is longer at early times weibull_shape = 1.5 # mean of Weibull = η * Γ(1 + 1/shape) import math eta = (1.0 / mean_sdc) / math.gamma(1.0 + 1.0 / weibull_shape) # Median lifetime: η * (ln 2)^(1/shape) lifetime_h = eta * (math.log(2.0) ** (1.0 / weibull_shape)) lifetime_model = f"Weibull(shape={weibull_shape:.1f}, η={eta:.0f}h)" else: lifetime_h = float("inf") lifetime_model = "Weibull (no failures)" return MCReliabilityResult( mean_reliability = float(np.mean(rel_arr)), std_reliability = float(np.std(rel_arr)), ci_low = ci_lo, ci_high = ci_hi, mean_sdc_per_hr = mean_sdc, p99_sdc_per_hr = float(np.percentile(sdc_arr, 99)), expected_mission_lifetime_h = lifetime_h, num_samples = num_samples, lifetime_model = lifetime_model, )
